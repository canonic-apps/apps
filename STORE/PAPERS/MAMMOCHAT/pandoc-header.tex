% MammoChat TYPESET header for two-column PDF
% Convert longtable to tabular in twocolumn mode

\usepackage{longtable}
\usepackage{supertabular}

\makeatletter
\if@twocolumn
  \renewenvironment{longtable}[2][c]{%
    \par\vspace{6pt}%
    \begin{center}\footnotesize%
    \begin{tabular}{#2}%
  }{%
    \end{tabular}%
    \end{center}%
    \par\vspace{6pt}%
  }
  \renewcommand{\endhead}{}
  \renewcommand{\endlastfoot}{}
  \renewcommand{\endfirsthead}{}
  \renewcommand{\endfoot}{}
\fi
\makeatother
